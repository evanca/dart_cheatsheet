Dart DSA Cheatsheet

A concise, beautifully formatted reference for Dart algorithms, collections, and data structures.

dart:core & dart:collection

Lists (Dynamic Arrays)

var list = <int>[1, 2, 3];

// Initialization
var zeros = List.filled(5, 0);
var grid = List.generate(3, (_) => List.filled(3, 0));

// O(1) Operations
list.add(4);          // Append
list.last;            // Peek end
list.removeLast();    // Pop end

// O(N) Operations
list.insert(0, 99);   // Insert at given index
list.removeAt(0);     // Remove at index
list.indexOf(2);      // Find index of element

// Spread & Destructuring
var combined = [...list, 4, 5];

// Iteration (O(N))
for (var item in list) { }
list.forEach((item) => print(item));

Maps (Hash Tables)

var map = <String, int>{'a': 1};

// O(1) Avg Operations
map['b'] = 2;         // Insert or Update
map.remove('a');      // Delete
map.containsKey('b'); // Quick check

// Advanced Insertion
map.putIfAbsent('c', () => 3);
map.update('b', (v) => v + 1, ifAbsent: () => 1);

// Iteration (O(N))
for (var entry in map.entries) {
  print('\${entry.key}: \${entry.value}');
}

Sets (Hash Sets)

var set = <int>{1, 2, 3};

// O(1) Avg Operations
set.add(4);
set.remove(2);
set.contains(3);      // Extremely fast lookup

// O(N) Set Mathematics
var a = {1, 2}, b = {2, 3};
a.union(b);           // {1, 2, 3}
a.intersection(b);    // {2}
a.difference(b);      // {1}

Stacks (LIFO)

Dart uses List. Never insert at index 0.

var stack = <int>[];

// Push (O(1))
stack.add(1);
stack.add(2);

// Pop (O(1))
if (stack.isNotEmpty) {
  var top = stack.removeLast(); // returns 2
}

// Peek (O(1))
var peek = stack.last;          // gives 1

Queues (FIFO & Deque)

Uses dart:collection. Circular buffer backed.

import 'dart:collection';

var queue = Queue<int>(); // A DoubleLinkedQueue

// FIFO: Enqueue & Dequeue (O(1))
queue.addLast(1);     // Alias: queue.add()
if (queue.isNotEmpty) {
  var front = queue.removeFirst();
}

// Deque: Double-Ended Operations (O(1))
queue.addFirst(0);    // Push front
queue.removeLast();   // Pop back
queue.first;          // Peek front
queue.last;           // Peek back

Trees & Sorted Collections

Self-balancing binary trees

import 'dart:collection';

// Sorted Map (SplayTreeMap) - O(log N)
var treeMap = SplayTreeMap<int, String>();
treeMap[3] = 'C';
treeMap[1] = 'A';
print(treeMap.firstKey()); // Iterates pre-sorted: 1

// Sorted Set (SplayTreeSet) - O(log N)
var treeSet = SplayTreeSet<int>();
treeSet.addAll([5, 1, 8]);
print(treeSet.first); // Returns 1 optimally

Math & Operators

// Truncating Division & Modulo
var div = 5 ~/ 2;     // 2 (Integer div)
var mod = 5 % 2;      // 1 (Remainder)

// Properties
print(5.isEven);      // false
print(5.isOdd);       // true
print((-5).abs());    // 5

// Bitwise AND (&) - 1 if both bits are 1
// 5 = 101
// 1 = 001
// ---------
//     001  → 1
var and = 5 & 1;

// Bitwise OR (|) - 1 if either bit is 1
// 5 = 101
// 2 = 010
// ---------
//     111  → 7
var or = 5 | 2;

// Bitwise XOR (^) - 1 if bits are different
// 5 = 101
// 1 = 001
// ---------
//     100  → 4
var xor = 5 ^ 1;

// Shift Left (<<) - Shifts bits left by n
// 1 = 001
// << 2
// ---------
//     100  → 4
var shift = 1 << 2;

Recursion & Memoization

var memo = <int, int>{};

int fib(int n) {
  if (n <= 1) return n;
  if (memo.containsKey(n)) return memo[n]!;
  return memo[n] = fib(n - 1) + fib(n - 2);
}

Graph Traversal (BFS / DFS)

var graph = <int, List<int>>{
  1: [2, 3], 2: [4], 3: [], 4: []
};

// BFS (Queue)
void bfs(int start) {
  var q = Queue<int>()..add(start);
  var vis = <int>{start};
  
  while (q.isNotEmpty) {
    var node = q.removeFirst();
    for (var next in graph[node] ?? []) {
      if (vis.add(next)) q.add(next);
    }
  }
}

// DFS (Recursive)
var vis = <int>{};
void dfs(int node) {
  if (!vis.add(node)) return;
  for (var next in graph[node] ?? []) dfs(next);
}
