// Recursive Grid DFS
var vis = <(int, int)>{};
void dfs(int r, int c) {
  if (r < 0 || r >= R || c < 0 || c >= C || !vis.add((r, c))) return;
  for (var d in dirs) dfs(r + d[0], c + d[1]);
}

// Iterative Grid BFS (Queue)
void bfs(int startR, int startC) {
  var q = Queue<(int, int)>()..add((startR, startC));
  var vis = { (startR, startC) };
  while (q.isNotEmpty) {
    var (r, c) = q.removeFirst();
    for (var d in dirs) {
      var nr = r + d[0], nc = c + d[1];
      if (nr >= 0 && nr < R && nc >= 0 && nc < C && vis.add((nr, nc))) {
        q.add((nr, nc));
      }
    }
  }
}

Quicksort (Divide & Conquer)

Recursive sort. O(N log N) avg, O(N²) worst.

List<int> quickSort(List<int> arr) {
  if (arr.length <= 1) return arr;
  
  var pivot = arr[arr.length ~/ 2];
  var left = arr.where((x) => x < pivot).toList();
  var middle = arr.where((x) => x == pivot).toList();
  var right = arr.where((x) => x > pivot).toList();
  
  return [...quickSort(left), ...middle, ...quickSort(right)];
}

Binary Search (Divide & Conquer)

Search sorted list. O(log N).

int binarySearch(List<int> sortedList, int target) {
  int low = 0, high = sortedList.length - 1;
  
  while (low <= high) {
    int mid = low + (high - low) ~/ 2;
    if (sortedList[mid] == target) return mid;
    if (sortedList[mid] < target) low = mid + 1;
    else high = mid - 1;
  }
  return -1; // Not found
}

Reverse Linked List

Classic pointer manipulation (Iterative)

ListNode? reverseList(ListNode? head) {
  ListNode? prev;
  ListNode? curr = head;
  
  while (curr != null) {
    ListNode? nextTemp = curr.next;
    curr.next = prev; // Reverse link
    prev = curr;
    curr = nextTemp;
  }
  return prev; // New head
}

Graph Traversal (BFS / DFS)

var graph = <int, List<int>>{
  1: [2, 3], 2: [4], 3: [], 4: []
};

// BFS (Queue)
void bfs(int start) {
  var q = Queue<int>()..add(start);
  var vis = <int>{start};
  
  while (q.isNotEmpty) {
    var node = q.removeFirst();
    for (var next in graph[node] ?? []) {
      if (vis.add(next)) q.add(next);
    }
  }
}

// DFS (Recursive)
var vis = <int>{};
void dfs(int node) {
  if (!vis.add(node)) return;
  for (var next in graph[node] ?? []) dfs(next);
}

Topological Sort

// Kahn with BFS:

List<String> topoSort(Map<String, List<String>> graph) {
  final indegree = <String, int>{};

  // Initialize indegree
  for (var node in graph.keys) {
    indegree[node] = 0;
  }

  for (var neighbors in graph.values) {
    for (var n in neighbors) {
      indegree[n] = (indegree[n] ?? 0) + 1;
    }
  }

  // Queue of nodes with indegree 0
  final queue = <String>[];
  for (var entry in indegree.entries) {
    if (entry.value == 0) queue.add(entry.key);
  }

  final result = <String>[];

  while (queue.isNotEmpty) {
    final node = queue.removeAt(0);
    result.add(node);

    for (var neighbor in graph[node] ?? []) {
      indegree[neighbor] = indegree[neighbor]! - 1;
      if (indegree[neighbor] == 0) {
        queue.add(neighbor);
      }
    }
  }

  return result;
}

DFS with Backtracking

// Generate all city combinations using DFS with backtracking
List<List<String>> cityPlans(List<String> cities) {
  // Result stores all built combinations
  final result = <List<String>>[];
  final path = <String>[];

  void dfs(int start) {
    result.add(path.toList());

    for (var i = start; i < cities.length; i++) {
      // Choose next city
      path.add(cities[i]);
      // Explore next choices
      dfs(i + 1);
      // Un-choose to backtrack
      path.removeLast();
    }
  }

  dfs(0);
  return result;
}

final plans = cityPlans(['A', 'B', 'C']);
// Plans:
// [], [A], [A,B], [A,B,C], [A,C], [B], [B,C], [C]

Leetcode 295. Find Median from Data Stream

import 'package:collection/collection.dart';

class MedianFinder {
  // Max heap via negated values
  final PriorityQueue<int> left = PriorityQueue<int>(); 
  // Min heap
  final PriorityQueue<int> right = PriorityQueue<int>(); 

  int get _leftLen => left.length;
  int get _rightLen => right.length;

  void addNum(int num) {
    left.add(-num);

    if (_leftLen > 0 && _rightLen > 0 && -left.first > right.first) {
      right.add(-left.removeFirst());
    }

    if (_leftLen > _rightLen + 1) {
      right.add(-left.removeFirst());
    }

    if (_rightLen > _leftLen + 1) {
      left.add(-right.removeFirst());
    }
  }

  double findMedian() {
    if (_leftLen == 0 && _rightLen == 0) {
      throw StateError('Cannot find median of an empty stream.');
    }

    if (_leftLen == _rightLen) {
      return (-left.first + right.first) / 2.0;
    }

    return _leftLen > _rightLen
        ? -left.first.toDouble()
        : right.first.toDouble();
  }
}

LeetCode 743. Network Delay Time

import 'package:collection/collection.dart';

class Solution {
  int networkDelayTime(List<List<int>> times, int n, int k) {
    // graph[u] contains all outgoing edges from node u.
    // Example: graph[2] = [(to: 1, weight: 1), (to: 3, weight: 4)]
    final Map<int, List<({int to, int weight})>> graph = {};

    for (final edge in times) {
      // Each edge is [from, to, time].
      // Example: [2, 1, 1] means 2 -> 1 takes 1 unit of time.
      final [u, v, w] = edge;

      graph.putIfAbsent(u, () => []);
      graph[u]!.add((to: v, weight: w));
    }

    // Min-heap gives us the node with the smallest distance first.
    final minHeap = PriorityQueue<({int dist, int node})>(
      (a, b) => a.dist.compareTo(b.dist),
    );

    minHeap.add((dist: 0, node: k));

    // Once a node is visited, its shortest distance is final.
    final visited = <int>{};

    int maxTime = 0;

    while (minHeap.isNotEmpty) {
      final current = minHeap.removeFirst();

      final int node = current.node;
      final int dist = current.dist;

      if (visited.contains(node)) continue;

      visited.add(node);

      // The answer is the largest shortest time.
      maxTime = maxTime > dist ? maxTime : dist;

      // If this node has no outgoing edges, use an empty list.
      for (final edge in graph[node] ?? <({int to, int weight})>[]) {
        final int next = edge.to;
        final int weight = edge.weight;

        if (!visited.contains(next)) {
          minHeap.add((dist: dist + weight, node: next));
        }
      }
    }

    return visited.length == n ? maxTime : -1;
  }
}
