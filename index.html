Maps (Hash Tables)

var map = <String, int>{'a': 1};

// O(1) Avg Operations
map['b'] = 2;         // Insert or Update
map.remove('a');      // Delete
map.containsKey('b'); // Quick check

// Advanced Insertion
map.putIfAbsent('c', () => 3);
map.update('b', (v) => v + 1, ifAbsent: () => 1);

// Iteration (O(N))
for (var entry in map.entries) {
  print('\${entry.key}: \${entry.value}');
}

Stacks (LIFO)

Dart uses List. Never insert at index 0.

var stack = <int>[];

// Push (O(1))
stack.add(1);
stack.add(2);

// Pop (O(1))
if (stack.isNotEmpty) {
  var top = stack.removeLast(); // returns 2
}

// Peek (O(1))
var peek = stack.last;          // gives 1

Queues (FIFO & Deque)

Uses dart:collection. Circular buffer backed.

import 'dart:collection';

var queue = Queue<int>(); // A DoubleLinkedQueue

// FIFO: Enqueue & Dequeue (O(1))
queue.addLast(1);     // Alias: queue.add()
if (queue.isNotEmpty) {
  var front = queue.removeFirst();
}

// Deque: Double-Ended Operations (O(1))
queue.addFirst(0);    // Push front
queue.removeLast();   // Pop back
queue.first;          // Peek front
queue.last;           // Peek back

Priority Queue (Heaps)

Uses package:collection. O(log N) inserts/removals.

import 'package:collection/collection.dart';

// Min-Heap (Default)
var minHeap = PriorityQueue<int>();
minHeap.add(5);
minHeap.add(1);
print(minHeap.first);       // 1 (Peek)
print(minHeap.removeFirst()); // 1 (Pop)

// Max-Heap (via -1 Hack for numericals)
var maxQ = PriorityQueue<int>();
maxQ.addAll([-3, -8, -2]);
print(-maxQ.removeFirst()); // 8

Quicksort (Divide & Conquer)

Efficient, recursive sort. O(N log N) average.

List<int> quickSort(List<int> arr) {
  if (arr.length <= 1) return arr;
  
  var pivot = arr[arr.length ~/ 2];
  var left = arr.where((x) => x < pivot).toList();
  var middle = arr.where((x) => x == pivot).toList();
  var right = arr.where((x) => x > pivot).toList();
  
  return [...quickSort(left), ...middle, ...quickSort(right)];
}

Binary Search (Divide & Conquer)

Efficiently find an element in a sorted list. O(log N).

int binarySearch(List<int> sortedList, int target) {
  int low = 0, high = sortedList.length - 1;
  
  while (low <= high) {
    int mid = low + (high - low) ~/ 2;
    if (sortedList[mid] == target) return mid;
    if (sortedList[mid] < target) low = mid + 1;
    else high = mid - 1;
  }
  return -1; // Not found
}

Reverse Linked List

Classic pointer manipulation (Iterative)

ListNode? reverseList(ListNode? head) {
  ListNode? prev;
  ListNode? curr = head;
  
  while (curr != null) {
    ListNode? nextTemp = curr.next;
    curr.next = prev; // Reverse link
    prev = curr;
    curr = nextTemp;
  }
  return prev; // New head
}

Recursion & Memoization

var memo = <int, int>{};

int fib(int n) {
  if (n <= 1) return n;
  if (memo.containsKey(n)) return memo[n]!;
  return memo[n] = fib(n - 1) + fib(n - 2);
}

Graph Traversal (BFS / DFS)

var graph = <int, List<int>>{
  1: [2, 3], 2: [4], 3: [], 4: []
};

// BFS (Queue)
void bfs(int start) {
  var q = Queue<int>()..add(start);
  var vis = <int>{start};
  
  while (q.isNotEmpty) {
    var node = q.removeFirst();
    for (var next in graph[node] ?? []) {
      if (vis.add(next)) q.add(next);
    }
  }
}

// DFS (Recursive)
var vis = <int>{};
void dfs(int node) {
  if (!vis.add(node)) return;
  for (var next in graph[node] ?? []) dfs(next);
}
